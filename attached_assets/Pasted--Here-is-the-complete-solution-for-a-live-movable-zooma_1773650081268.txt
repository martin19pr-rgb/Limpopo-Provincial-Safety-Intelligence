**Here is the complete solution for a live, movable, zoomable provincial map background** in the Premier’s Office Dashboard.

This map covers the **entire Limpopo province** by default and allows smooth zooming all the way down to individual cities, towns, and roads. The Premier can see **exactly what is transpiring** in real time (incidents, officers, sensors, traffic lights, weather impact, citizen SOS, etc.) and configure alerts directly on the map.

### SmartProvincialMap.tsx (Copy-Paste This File)

Create a new file: `components/SmartProvincialMap.tsx`

```tsx
'use client';

import { useEffect, useRef } from 'react';
import L from 'leaflet';
import 'leaflet/dist/leaflet.css';
import { useMapEvents } from 'react-leaflet';
import { motion } from 'framer-motion';

// Fix Leaflet marker icons in Vite/Next.js
delete (L.Icon.Default.prototype as any)._getIconUrl;
L.Icon.Default.mergeOptions({
  iconRetinaUrl: 'https://unpkg.com/leaflet@1.9.4/dist/images/marker-icon-2x.png',
  iconUrl: 'https://unpkg.com/leaflet@1.9.4/dist/images/marker-icon.png',
  shadowUrl: 'https://unpkg.com/leaflet@1.9.4/dist/images/marker-shadow.png',
});

export default function SmartProvincialMap() {
  const mapRef = useRef<L.Map | null>(null);
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!containerRef.current || mapRef.current) return;

    const map = L.map(containerRef.current, {
      zoomControl: true,
      attributionControl: true,
    }).setView([-23.5, 29.5], 8); // Default: full Limpopo

    mapRef.current = map;

    // Base tile layer
    L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '© OpenStreetMap contributors',
    }).addTo(map);

    // Limpopo district boundaries (you can replace with real GeoJSON later)
    // For now, this creates a placeholder outline
    const limpopoBounds = L.polygon([
      [-22.5, 28.0], [-22.5, 31.0], [-25.0, 31.0], [-25.0, 28.0]
    ], { color: '#00BFFF', weight: 3, fillOpacity: 0.05 }).addTo(map);

    // Cleanup
    return () => {
      if (mapRef.current) {
        mapRef.current.remove();
        mapRef.current = null;
      }
    };
  }, []);

  // Live data layers will be added here in the next step

  return (
    <div ref={containerRef} className="w-full h-screen rounded-3xl overflow-hidden border border-gray-700 relative">
      {/* Overlay controls */}
      <div className="absolute top-4 right-4 z-[1000] flex flex-col gap-3">
        <motion.button
          whileHover={{ scale: 1.05 }}
          whileTap={{ scale: 0.95 }}
          className="glassmorphic px-6 py-3 text-sm font-medium"
          onClick={() => mapRef.current?.setView([-23.5, 29.5], 8)}
        >
          Reset to Province
        </motion.button>
      </div>

      {/* Live status indicator */}
      <div className="absolute bottom-4 left-4 z-[1000] bg-black/70 backdrop-blur-md px-4 py-2 rounded-2xl text-xs text-green-400 flex items-center gap-2">
        <div className="w-2 h-2 bg-green-400 rounded-full animate-pulse" />
        LIVE • Limpopo Province
      </div>
    </div>
  );
}
```
